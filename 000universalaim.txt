-- Function to aim at target
local function aimbot(targetPlayer)
    if targetPlayer.Character and targetPlayer.Character:FindFirstChild("HumanoidRootPart") then
        local aimPart = targetPlayer.Character.HumanoidRootPart
        local targetPos = aimPart.Position
        local cameraPos = Camera.CFrame.Position
        local direction = (targetPos - cameraPos).unit
        Camera.CFrame = CFrame.new(cameraPos, cameraPos + direction)
    end
end

-- Function to check if player is on same team
local function isSameTeam(player)
    if not settings.teamCheckEnabled then
        return false
    end
    
    local localTeam = localPlayer.Team
    local playerTeam = player.Team
    
    if localTeam and playerTeam and localTeam == playerTeam then
        return true
    end
    
    return false
end

-- Function to get health color (green to red)
local function getHealthColor(health, maxHealth)
    local healthPercent = math.max(0, math.min(1, health / maxHealth))
    
    if healthPercent >= 0.5 then
        local ratio = (healthPercent - 0.5) * 2
        return Color3.fromRGB(255 * (1 - ratio), 255, 0)
    else
        local ratio = healthPercent * 2
        return Color3.fromRGB(255, 255 * ratio, 0)
    end
end
